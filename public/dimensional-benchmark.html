class Point {
  constructor(name) {
    this.name = name;
    this.dimensions = {};
    this.tether = [];
  }
  observe(dimension, context={}) {
    if (!this.dimensions[dimension]) {
      this.dimensions[dimension] = this.manifest(dimension, context);
      this.tether.push(`${this.name}->${dimension}`);
    }
    return this.dimensions[dimension];
  }
  manifest(dim, ctx) {
    // In production: this is where you'd call LLM with tethered context
    const db = {
      engine: {type:"electric", battery:"12%", status:"no_start"},
      seats: {count:5, material:"vegan_leather"},
      color: "red", make: "Tesla", model: "3", year: 2023,
      radio: {presets:[], volume:0}, gps: {lat:40.1,lon:-111.6}
    };
    return db[dim] || null;
  }
  toContext() {
    // Only serialize manifested dimensions
    return {point: this.name, tether: this.tether, ...this.dimensions};
  }
}
